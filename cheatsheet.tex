\section{Recursive search for element in a list}

\begin{lstlisting}[language=Python]
def indexof(num, lst):
    if len(lst) == 0:
        return -1
    if lst[0] == num:
        return 0
    index = indexof(num, lst[1:])
    if index != -1:
        index += 1
    return index
\end{lstlisting}
